121 Colum[2][0] = -sin(angle);
122 Colum[0][2] = sin(angle);

246//Returns the Determinant of Matrix Matrix
247float FMatrix4::Det()
248{
249 float calculation, result = 1;
250 FMatrix4 tempMatrix = *this;
251
252 for (int i = 0; i < 4; i++)
253 {
254 for (int k = i + 1; k < 4; k++)
255 {
256 calculation = tempMatrix[k][i] / tempMatrix[i][i];
257
258 for (int j = i; j < 4; j++)
259 {
260 tempMatrix[k][j] = tempMatrix[k][j] - calculation * tempMatrix[i][j];
261 }
262 }
263 }
264
265 for (int i = 0; i < 4; i++) result *= tempMatrix[i][i];
266 return result;
267}

551FMatrix4 FMatrix4::GetRotationMatrix(float angle, float x, float y, float z, bool isAngleRadian)
552{
553 if (!isAngleRadian) angle *= M_PI/180.0f;//Converts input from degrees to radians
554
555 float cosang, sinang, cosm;
556 FVector3 rotAxis(x, y, z);
557 rotAxis.Normalize();
558 cosang = cos(angle);
559 sinang = sin(angle);
560 cosm = (1.0f - cosang);
561
562 FMatrix4 returnMatrix;
563
564 returnMatrix[0][0] = (rotAxis.X * rotAxis.X * cosm) + cosang;
565 returnMatrix[0][1] = (rotAxis.X * rotAxis.Y * cosm) + (rotAxis.Z * sinang);
566 returnMatrix[0][2] = (rotAxis.X * rotAxis.Z * cosm) - (rotAxis.Y * sinang);
567 returnMatrix[0][3] = 0.0;
568
569 returnMatrix[1][0] = (rotAxis.X * rotAxis.Y * cosm) - (rotAxis.Z * sinang);
570 returnMatrix[1][1] = (rotAxis.Y * rotAxis.Y * cosm) + cosang;
571 returnMatrix[1][2] = (rotAxis.Y * rotAxis.Z * cosm) + (rotAxis.X * sinang);
572 returnMatrix[1][3] = 0.0;
573
574 returnMatrix[2][0] = (rotAxis.X * rotAxis.Z * cosm) + (rotAxis.Y * sinang);
575 returnMatrix[2][1] = (rotAxis.Y * rotAxis.Z * cosm) - (rotAxis.X * sinang);
576 returnMatrix[2][2] = (rotAxis.Z * rotAxis.Z * cosm) + cosang;
577 returnMatrix[2][3] = 0.0;
578
579 returnMatrix[3][0] = 0.0;
580 returnMatrix[3][1] = 0.0;
581 returnMatrix[3][2] = 0.0;
582 returnMatrix[3][3] = 1.0;
583
584 return returnMatrix;
585}

619FMatrix4 FMatrix4::GetLookAtMatrix(const FVector3& eye, const FVector3& target, const FVector3& up)
620{
621 FMatrix4 returnMatrix;
622 FVector3 forward = (target - eye);
623 forward.Normalize();
624 FVector3 right = forward.CrossProduct(up).GetNormal();
625 FVector3 upVec = right.CrossProduct(forward).GetNormal();
626
627 returnMatrix[0][0] = right.X;
628 returnMatrix[1][0] = right.Y;
629 returnMatrix[2][0] = right.Z;
630
631 returnMatrix[0][1] = upVec.X;
632 returnMatrix[1][1] = upVec.Y;
633 returnMatrix[2][1] = upVec.Z;
634
635 returnMatrix[0][2] = -forward.X;
636 returnMatrix[1][2] = -forward.Y;
637 returnMatrix[2][2] = -forward.Z;
638
639 returnMatrix[3][0] = -(right * eye);
640 returnMatrix[3][1] = -(upVec * eye);
641 returnMatrix[3][2] = (forward * eye);
642
643 return returnMatrix;

651FMatrix4 FMatrix4::GetPerspectiveMatrix(float fieldOfView, float aspect, float zNear, float zFar, bool isRadian)
652{
653 if (!isRadian) fieldOfView *= M_PI / 180.0f;
654
655 float const tanHalfFovy = tan(fieldOfView / 2);
656 FMatrix4 returnMatrix = FMatrix4(0);
657
658 returnMatrix[0][0] = 1.0f / (aspect * tanHalfFovy);
659 returnMatrix[1][1] = 1.0f / (tanHalfFovy);
660 returnMatrix[2][2] = - (zFar + zNear) / (zFar - zNear);
661 returnMatrix[2][3] = - 1.0f;
662 returnMatrix[3][2] = - (2 * zFar * zNear) / (zFar - zNear);
663
664 return returnMatrix;
