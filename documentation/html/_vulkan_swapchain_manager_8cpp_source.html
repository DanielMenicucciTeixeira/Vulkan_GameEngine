779void VulkanSwapchainManager::TransitionImageLayout(VkImage_T* image, VkFormat format, VkImageLayout oldLayout, VkImageLayout newLayout, int layerCount)
780{
781 VkCommandBuffer commandBuffer = Manager->BeginSingleTimeCommands();
782
783 VkImageMemoryBarrier barrier{};
784 barrier.sType = VK_STRUCTURE_TYPE_IMAGE_MEMORY_BARRIER;
785 barrier.oldLayout = oldLayout;
786 barrier.newLayout = newLayout;
787 barrier.srcQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
788 barrier.dstQueueFamilyIndex = VK_QUEUE_FAMILY_IGNORED;
789 barrier.image = image;
790 barrier.subresourceRange.baseMipLevel = 0;
791 barrier.subresourceRange.levelCount = 1;
792 barrier.subresourceRange.baseArrayLayer = 0;
793 barrier.subresourceRange.layerCount = layerCount;
794
795 if (newLayout == VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL)
796 {
797 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_DEPTH_BIT;
798
799 if (HasStencilComponent(format))
800 {
801 barrier.subresourceRange.aspectMask |= VK_IMAGE_ASPECT_STENCIL_BIT;
802 }
803 }
804 else
805 {
806 barrier.subresourceRange.aspectMask = VK_IMAGE_ASPECT_COLOR_BIT;
807 }
808
809 VkPipelineStageFlags sourceStage;
810 VkPipelineStageFlags destinationStage;
811
812 if (oldLayout == VK_IMAGE_LAYOUT_UNDEFINED && newLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL)
813 {
814 barrier.srcAccessMask = 0;
815 barrier.dstAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
816
817 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
818 destinationStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
819 }
820 else if (oldLayout == VK_IMAGE_LAYOUT_TRANSFER_DST_OPTIMAL && newLayout == VK_IMAGE_LAYOUT_SHADER_READ_ONLY_OPTIMAL)
821 {
822 barrier.srcAccessMask = VK_ACCESS_TRANSFER_WRITE_BIT;
823 barrier.dstAccessMask = VK_ACCESS_SHADER_READ_BIT;
824
825 sourceStage = VK_PIPELINE_STAGE_TRANSFER_BIT;
826 destinationStage = VK_PIPELINE_STAGE_FRAGMENT_SHADER_BIT;
827 }
828 else if (oldLayout == VK_IMAGE_LAYOUT_UNDEFINED && newLayout == VK_IMAGE_LAYOUT_DEPTH_STENCIL_ATTACHMENT_OPTIMAL)
829 {
830 barrier.srcAccessMask = 0;
831 barrier.dstAccessMask = VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_READ_BIT | VK_ACCESS_DEPTH_STENCIL_ATTACHMENT_WRITE_BIT;
832
833 sourceStage = VK_PIPELINE_STAGE_TOP_OF_PIPE_BIT;
834 destinationStage = VK_PIPELINE_STAGE_EARLY_FRAGMENT_TESTS_BIT;
835 }
836 else
837 {
838 throw std::invalid_argument("unsupported layout transition!");
839 }
840
841 vkCmdPipelineBarrier(commandBuffer, sourceStage, destinationStage, 0, 0, nullptr, 0, nullptr, 1, &barrier);
842
843 Manager->EndSingleTimeCommands(commandBuffer);
844}
