101bool C_CameraComponent::FrustumCheck(C_BoundingBox* meshBox)
102{
103 bool wrongSide, rightSide;
104 int i, j, k;
105 FVector3 point;
106 bool getOut;
107
108 for (int p = 0; p < 6; p++) //for each plane
109 {
110 wrongSide = false; rightSide = false;
111 getOut = false;
112
113 for (i = 0; i < 2 && !getOut; i++)
114 {
115 if (i == 0) point.X = meshBox->GetMin().X;
116 else point.X = meshBox->GetMax().X;
117
118 for (j = 0; j < 2 && !getOut; j++)
119 {
120 if (j == 0) point.Y = meshBox->GetMin().Y;
121 else point.Y = meshBox->GetMax().Y;
122
123 for (k = 0; k < 2 && !getOut; k++)
124 {
125 if (k == 0) point.Z = meshBox->GetMin().Z;
126 else point.Z = meshBox->GetMax().Z;
127
128 // is the corner outside or inside
129 if (Frustum[p] * FVector4(point.X, point.Y, point.Z, 1) < 0) wrongSide = true;
130 else rightSide = true;
131
132 //get out of the cycle as soon as a box has corners both inside and out of the plane
133 getOut = (rightSide == true && wrongSide == true);
134 }
135 }
136 }
137
138 if (!rightSide) return (false);//if all corners are out
139 }
140 return(true);
141}
