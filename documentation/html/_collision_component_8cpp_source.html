139bool C_CollisionComponent::GJK(const C_CollisionComponent* colliderA, const C_CollisionComponent* colliderB)
140{
141 // Get initial support point in any direction
142 FVector3 support = GetFurthestMinkDiffPoint(colliderA, colliderB, FVector3(0.0f, 0.0f, 1.0f));
143
144 // Simplex is an array of points, max count is 4
145 Simplex points;
146 points.PushFront(support);
147
148 // New direction is towards the origin
149 FVector3 direction = -support;
150 support = GetFurthestMinkDiffPoint(colliderA, colliderB, direction);
151
152 while (true)
153 {
154 if (support * direction <= 0)
155 {
156 return false; // no collision
157 }
158
159 Line(points, direction);
160 points.PushFront(support);
161 Triangle(points, direction);
162 points.PushFront(support);
163 if (Tetrahedron(points, direction))
164 {
165 return true;
166 }
167 else
168 {
169 points.PushFront(support);
170 }
171 }
172}

196bool C_CollisionComponent::Line(Simplex& points, FVector3& direction)
197{
198 FVector3 a = points[0];
199 FVector3 b = points[1];
200
201 FVector3 ab = b - a;
202 FVector3 ao = -a;
203
204 if (SameDirection(ab, ao))
205 {
206 direction = ab.CrossProduct(ao).CrossProduct(ab);
207 }
208 else

210 points = { a };
211 direction = ao;

217bool C_CollisionComponent::Triangle(Simplex& points, FVector3& direction)
218{
219 FVector3 a = points[0];
220 FVector3 b = points[1];
221 FVector3 c = points[2];
222
223 FVector3 ab = b - a;
224 FVector3 ac = c - a;
225 FVector3 ao = -a;
226
227 FVector3 abc = ab.CrossProduct(ac);
228
229 if (SameDirection(abc.CrossProduct(ac), ao))
230 {
231 if (SameDirection(ac, ao)) {
232 points = { a, c };
233 direction = ac.CrossProduct(ao).CrossProduct(ac);
234 }
235
236 else
237 {
238 return Line(points = { a, b }, direction);
239 }
240 }
241
242 else
243 {
244 if (SameDirection(ab.CrossProduct(abc), ao))
245 {
246 return Line(points = { a, b }, direction);
247 }
248
249 else
250 {
251 if (SameDirection(abc, ao))
252 {
253 direction = abc;
254 }
255
256 else
257 {
258 points = { a, c, b };
259 direction = -abc;
260 }
261 }
262 }

264 return false;

267bool C_CollisionComponent::Tetrahedron(Simplex& points, FVector3& direction)
268{
269 FVector3 a = points[0];
270 FVector3 b = points[1];
271 FVector3 c = points[2];
272 FVector3 d = points[3];
273

275 FVector3 ac = c - a;
276 FVector3 ad = d - a;
277 FVector3 ao = -a;
278
279 FVector3 abc = ab.CrossProduct(ac);

281 FVector3 adb = ad.CrossProduct(ab);
282
283 if (SameDirection(abc, ao)) {
284 return Triangle(points = { a, b, c }, direction);
285 }
286
287 if (SameDirection(acd, ao)) {
288 return Triangle(points = { a, c, d }, direction);
289 }
290
291 if (SameDirection(adb, ao)) {
292 return Triangle(points = { a, d, b }, direction);
293 }
294
295 return true;
