213void UI_Element::Attach(UI_Element* parent)
214{
215 if (!parent || Parent == parent) return;
216
217 if (Parent) Parent->Children.erase(this);
218 if (Children.count(parent))
219 {
220 if (Parent) parent->Attach(Parent);
221 else parent->Dettach();
222 }
223
224 Parent = parent;
225 Parent->Children.insert(this);
226 Position = Rect.GetScreenPosition() - Parent->GetPosition();
227 FVector2 parentExtent = Parent->GetExtent();
228 Extent = Rect.GetRectExtent() | FVector2(parentExtent.X, parentExtent.Y);
229}
