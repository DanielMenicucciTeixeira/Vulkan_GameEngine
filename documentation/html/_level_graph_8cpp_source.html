45void LevelGraph::AddObject(O_Object* gameObject)
46{
47 if (gameObject)
48 {
49 size_t classID = typeid(*gameObject).hash_code();
50 if (!GameObjectsByClass.count(classID)) GameObjectsByClass[classID] = std::set<O_Object*>();
51 GameObjectsByClass[classID].insert(gameObject);
52
53 for (const auto& tag : gameObject->GetTags())
54 {
55 if (!GameObjectsByTag.count(tag)) GameObjectsByTag[tag] = std::set<O_Object*>();
56 GameObjectsByTag[tag].insert(gameObject);
57 }
58
59 std::string name = gameObject->GetName();
60 if (gameObject->GetName() == "")
61 {
62 name = typeid(*gameObject).name();
63 name = name.substr(6);//Remove the "class " part of the typeid name
64 }
65
66 if (GameObjectsByName.count(name))
67 {
68 unsigned int i = 1;
69 std::string newName;
70 do
71 {
72 newName = name + "(" + std::to_string(i) + ")";
73 if (GameObjectsByName.count(newName)) i++;
74 else
75 {
76 name = newName;
77 break;
78 }
79 } while (true);
80 }
81 GameObjectsByName[name] = gameObject;
82 gameObject->SetName(name);
83 }
84}

286{

288 if (!textures[textureName])
