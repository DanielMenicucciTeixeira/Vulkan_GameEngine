32FQuaternion::FQuaternion(const FVector3& euler, bool inRadians)
33{
34 FVector3 vector;
35 if (!inRadians) vector = euler * M_PI / 180;
36 else vector = euler;
37
38 // Abbreviations for the various angular functions
39 double cy = cos(vector.Z * 0.5);
40 double sy = sin(vector.Z * 0.5);
41 double cp = cos(vector.Y * 0.5);
42 double sp = sin(vector.Y * 0.5);
43 double cr = cos(vector.X * 0.5);
44 double sr = sin(vector.X * 0.5);
45
46 W = cr * cp * cy + sr * sp * sy;
47 X = sr * cp * cy - cr * sp * sy;
48 Y = cr * sp * cy + sr * cp * sy;
49 Z = cr * cp * sy - sr * sp * cy;
50}

53{

105FQuaternion FQuaternion::SLerp(FQuaternion quaternion0, FQuaternion quaternion1, float deltaTime)
106{
107 // Only unit quaternions are valid rotations.
108 // Normalize to avoid undefined behavior.
109 quaternion0.Normalize();
110 quaternion1.Normalize();
111
112 // Compute the cosine of the angle between the two vectors.
113 float dot = DotProduct(quaternion0, quaternion1);
114
115 // If the dot product is negative, slerp won't take
116 // the shorter path. Note that v1 and -v1 are equivalent when
117 // the negation is applied to all four components. Fix by
118 // reversing one quaternion.
119 if (dot < 0.0f) {
120 quaternion1 = quaternion1 * -1;
121 dot = -dot;
122 }
123
124 const float DOT_THRESHOLD = 0.9995;
125 if (dot > DOT_THRESHOLD) {
126 // If the inputs are too close for comfort, linearly interpolate
127 // and normalize the result.
128
129 FQuaternion result = quaternion0 + (quaternion1 - quaternion0) * deltaTime;
130 result;
131 return result;
132 }
133
134 // Since dot is in range [0, DOT_THRESHOLD], acos is safe
135 float theta0 = acos(dot); // theta_0 = angle between input vectors
136 float theta = theta0 * deltaTime; // theta = angle between v0 and result
137 float sinTheta = sin(theta); // compute this value only once
138 float sinTheta0 = sin(theta0); // compute this value only once
139
140 float scalar0 = cos(theta) - dot * sinTheta / sinTheta0; // == sin(theta_0 - theta) / sin(theta_0)
141 float scalar1 = sinTheta / sinTheta0;
142 FQuaternion result = (quaternion0 * scalar0) + (quaternion1 * scalar1);
143 return ((quaternion0 * scalar0) + (quaternion1 * scalar1));
144}
