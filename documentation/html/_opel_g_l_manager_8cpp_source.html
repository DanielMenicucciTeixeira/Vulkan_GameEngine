203void OpenGLManager::HandleMaterial(Material* material, int currentBinding, const unsigned int& program)
204{
205 const auto& info = material->GetShaderVariablesInfo();
206 const auto& data = material->GetShaderVariablesData();
207 for (int i = 0; i < info.size(); i++)
208 {
209 switch (info[i].Type)
210 {
211 case E_ShaderVariableType::UNIFORM_BUFFER:
212 BindUniformBufferFromMaterial(info[i], data[i], currentBinding, program);
213 currentBinding++;
214 break;
215 case E_ShaderVariableType::COMBINED_IMAGE_SAMPLER:
216 BindTextureFromMaterial(info[i], data[i], program);
217 break;
218 default:
219 break;
220 }
221 }
222}
