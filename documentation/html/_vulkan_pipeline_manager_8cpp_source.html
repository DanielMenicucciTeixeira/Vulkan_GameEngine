268void VulkanPipelineManager::CreatePipelineFromVulkanMaterial(std::string shader, M_VulkanMaterial* material)
269{
270 //Create Shader Modules and Info
271 std::vector<VkShaderModule> shaderModules;
272 std::vector<VkPipelineShaderStageCreateInfo> shaderStages;
273
274 for (const auto& file : material->GetShaderInfo().ShaderFiles)
275 {
276 std::string filePath = DEFAULT_SHADER_PATH + file.second + DEFAULT_SHADER_EXT;
277 shaderModules.push_back(CreateShaderModule(ReadFile(filePath)));
278
279 VkPipelineShaderStageCreateInfo shaderStageInfo{};
280 shaderStageInfo.sType = VK_STRUCTURE_TYPE_PIPELINE_SHADER_STAGE_CREATE_INFO;
281 shaderStageInfo.stage = GetVulkanShaderStageFlag(file.first);
282 shaderStageInfo.module = *(shaderModules.end() - 1);
283 shaderStageInfo.pName = "main";
284
285 shaderStages.push_back(shaderStageInfo);
286 }
287 //----------------------------------------------------------------------------
288
289 //Set DescriptorSetLayout
290 VkDescriptorSetLayout temp = Manager->GetDescriptorSetLayoutsByShader()[shader];
291 material->SetDescriptorSetLayoutInPipelineLayout(temp);
292
293 if (vkCreatePipelineLayout(Manager->GetLogicalDevice(), material->GetPipelineLayoutCreateInfo(), nullptr, &PipelinesMap[material->GetShaderInfo().Name].second) != VK_SUCCESS)
294 {
295 throw std::runtime_error("failed to create pipeline layout!");
296 }
297 //-------------------------------------------------------------------------------
298
299 //Set Layout and Render Pass
300 material->FillGraphicsPipeline(Manager->GetRenderPass(), PipelinesMap[material->GetShaderInfo().Name].second, shaderStages.size(), shaderStages.data());
301 auto stages = material->GetGraphicsPipelineStages();
302
303 //Create Graphics Pipeline
304 VkGraphicsPipelineCreateInfo pipelineInfo{};
305 pipelineInfo.sType = VK_STRUCTURE_TYPE_GRAPHICS_PIPELINE_CREATE_INFO;
306 pipelineInfo.stageCount = shaderStages.size();
307 pipelineInfo.pStages = shaderStages.data();
308 pipelineInfo.pVertexInputState = stages.VertexInput;
309 pipelineInfo.pInputAssemblyState = stages.InputAssembly;
310 pipelineInfo.pViewportState = stages.Viewport;
311 pipelineInfo.pRasterizationState = stages.Rasterization;
312 pipelineInfo.pMultisampleState = stages.Multisample;
313 pipelineInfo.pDepthStencilState = stages.DepthStencil;
314 pipelineInfo.pColorBlendState = stages.ColorBlend;
315 pipelineInfo.pDynamicState = stages.DynamicStatesInfo;
316 pipelineInfo.layout = PipelinesMap[material->GetShaderInfo().Name].second;
317 pipelineInfo.renderPass = Manager->GetRenderPass();
318 pipelineInfo.subpass = 0;
319 pipelineInfo.basePipelineHandle = VK_NULL_HANDLE; // Optional
320 pipelineInfo.basePipelineIndex = -1; // Optional
321
322 if (vkCreateGraphicsPipelines(Manager->GetLogicalDevice(), VK_NULL_HANDLE, 1, &pipelineInfo, nullptr, &PipelinesMap[material->GetShaderInfo().Name].first) != VK_SUCCESS)
323 {
324 throw std::runtime_error("failed to create graphics pipeline!");
325 }
326
327 for (auto& module : shaderModules) vkDestroyShaderModule(Manager->GetLogicalDevice(), module, nullptr);
