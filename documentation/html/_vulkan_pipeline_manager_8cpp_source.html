227std::vector<char> VulkanPipelineManager::ReadFile(const std::string& filename)
228{
229 std::ifstream file(filename, std::ios::ate | std::ios::binary);
230
231 if (!file.is_open())
232 {
233 throw std::runtime_error("failed to open file!");
234 }
235
236 size_t fileSize = (size_t)file.tellg();
237 std::vector<char> buffer(fileSize);
238 file.seekg(0);
239 file.read(buffer.data(), fileSize);
240 file.close();
241
242 return buffer;
243}
