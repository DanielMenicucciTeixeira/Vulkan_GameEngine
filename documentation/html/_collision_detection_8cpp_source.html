54bool CollisionDetection::RaySphereIntersection(Ray ray, Sphere sphere)
55{
56 FVector3 l = sphere.position - ray.GetOrigin();
57 float tca = l.GetDotProduct(ray.GetDirection());
58 if (tca < 0) { return false; }
59
60 float d2 = l.GetDotProduct(l) - tca * tca;
61 float radius2 = sphere.radius * sphere.radius;
62 if (d2 > radius2) { return false; }
63
64 float thc = sqrt(radius2 - d2);
65 float t0 = tca - thc;
66 float t1 = tca + thc;
67
68
69 if (t0 > t1) { std::swap(t0, t1); }
70
71 if (t0 < 0) {
72 t0 = t1; // if t0 is negative, let's use t1 instead
73 if (t0 < 0) return false; // both t0 and t1 are negative
74 }
75
76
77
78 ray.SetIntersectDistance(t0);
79 collisionData.CollisionPoint = ray.GetOrigin() + t0;
80
