22void VulkanPipelineManager::CreateGraphicsPipelines()
23{
24 //Create standard Pipelines
25 for (const auto& shader : Manager->GetRenderData()->DataMapByShader)
26 {
27 Material* material = (*shader.second.begin()).first;
28 if (M_VulkanMaterial* vulkanMaterial = dynamic_cast<M_VulkanMaterial*>(material)) CreatePipelineFromVulkanMaterial(shader.first, vulkanMaterial);
29 else CreatePipelineFromGenericMaterial(shader.first, material);
30 }
31
32 //Create UI Pipelines
33 for (const auto& shader : Manager->GetRenderData()->UIMapByShader)
34 {
35 M_UI_Material* material = (*shader.second.begin()).first;
36 if (M_UI_VulkanMaterial* vulkanMaterial = dynamic_cast<M_UI_VulkanMaterial*>(material)) CreateUIPipelineFromVulkanMaterial(shader.first, vulkanMaterial);
37 else CreateUIPipelineFromGenericMaterial(shader.first, material);
38 }
39}

73std::vector<char> VulkanPipelineManager::ReadFile(const std::string& filename)
74{
75 std::ifstream file(filename, std::ios::ate | std::ios::binary);
76
77 if (!file.is_open())
78 {
79 throw std::runtime_error("failed to open file!");
80 }
81
82 size_t fileSize = (size_t)file.tellg();
83 std::vector<char> buffer(fileSize);
84 file.seekg(0);
85 file.read(buffer.data(), fileSize);
86 file.close();
87
88 return buffer;
89}

279{
280 //Create Shader Modules and Info

320 pipelineInfo.pViewportState = stages.Viewport;
321 pipelineInfo.pRasterizationState = stages.Rasterization;
322 pipelineInfo.pMultisampleState = stages.Multisample;
323 pipelineInfo.pDepthStencilState = stages.DepthStencil;
324 pipelineInfo.pColorBlendState = stages.ColorBlend;
325 pipelineInfo.pDynamicState = stages.DynamicStatesInfo;
326 pipelineInfo.layout = PipelinesMap[material->GetShaderInfo().Name].second;
327 pipelineInfo.renderPass = Manager->GetRenderPass();
328 pipelineInfo.subpass = 0;
329 pipelineInfo.basePipelineHandle = VK_NULL_HANDLE; // Optional
330 pipelineInfo.basePipelineIndex = -1; // Optional
331
332 if (vkCreateGraphicsPipelines(Manager->GetLogicalDevice(), VK_NULL_HANDLE, 1, &pipelineInfo, nullptr, &PipelinesMap[material->GetShaderInfo().Name].first) != VK_SUCCESS)
