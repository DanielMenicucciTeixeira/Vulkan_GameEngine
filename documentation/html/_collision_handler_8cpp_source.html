57S_CollisionData CollisionHandler::GetCollisionSingleRay(Ray& ray)
58{
59 S_CollisionData data;
60 float shortestDistance = FLT_MAX;
61 bool isCollideing = false;
62 for (auto coll : scenePartition->GetCollision(ray)) {
63 switch (coll->GetColliderType())
64 {
65 case ColliderType::BoundingBox:
66 isCollideing = CollisionDetection::RayAABBIntersection(ray, static_cast<C_BoundingBox*>(coll)->GetBoxBounds());
67 break;
68
69 case ColliderType::Sphere:
70 isCollideing = CollisionDetection::RaySphereIntersection(ray, static_cast<C_SphereCollider*>(coll)->GetCollisionSphere());
71 break;
72
73 case ColliderType::Box:
74 isCollideing = CollisionDetection::RayOBBIntersection(ray, static_cast<C_BoxCollider*>(coll)->GetCollisionBox());
75 break;
76 }
77
78 if (isCollideing) {
79 if (shortestDistance > ray.GetIntersectDistance()) {
80 data = CollisionDetection::GetCollisionData();
81 data.OtherCollisonComponent = coll;
82 shortestDistance = ray.GetIntersectDistance();
83 }
84 }
85 }
86 return data;
87}
