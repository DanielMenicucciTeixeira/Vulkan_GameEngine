207void C_PhysicsComponent::AABBResponse(C_BoundingBox* coll1, C_BoundingBox* coll2)
208{
209 //First determine which one is farther left.
210
211 //Is the first colldier staic
212 bool isFirstStatic = coll1->GetIsStatic();
213
214 //Is the second collider static
215 bool isSecondStatic = coll2->GetIsStatic();
216
217 //If both are static then neither can move. This should not happen becuase the static split check but just in case?
218 if (isFirstStatic && isSecondStatic) { return; }
219
220 FVector3 min1 = coll1->GetBoxBounds().GetPosition();
221 FVector3 min2 = coll2->GetBoxBounds().GetPosition();
222
223 FVector3 max1 = min1 + coll1->GetBoxBounds().GetExtent();
224 FVector3 max2 = min2 + coll2->GetBoxBounds().GetExtent();
225
226 FVector3 depthPenetration;
227
228 //three spliting points
229 //1. is coll1 to the left or right of the other.
230 //2. is coll2 static or not
231 //3. does coll2 have a parent or not.
232
233 //Notes
234 //if coll2 is static it does not matter if it has a parent or not.
235 //By making the depthPenetration negative if coll1 is on the right, the translation can be negated allowing for the bypassing of another nesting section.
236 //A collider without a object is by definition static. so 3 being false = 2
237
238 //Push X
239
240 if (min1.X < min2.X) {
241 depthPenetration.X = max1.X - min2.X;
242 }
243 else if (min1.X > min2.X) {
244 depthPenetration.X = max2.X - min1.X; }
245
246 //Push Y
247
248 if (min1.Y < min2.Y) {
249 depthPenetration.Y = max1.Y - min2.Y; }
250 else if (min1.Y > min2.Y) {
251 depthPenetration.Y = -(std::abs(max2.Y - min1.Y)); };
252
253 //Push Z
254
255 if (min1.Z < min2.Z) {
256 depthPenetration.Z = std::abs(max1.Z - min2.Z); }
257 else if (min1.Z > min2.Z) {
258 depthPenetration.Z = max2.Z - min1.Z;
259 }
260
261
262 //Push by the smallest distance.
263
264 //Solution to edge collision. (do wish I could figure out a better way but ehh for now)
265 depthPenetration += FVector3(0.0000000000001f);
266
267 depthPenetration = depthPenetration.GetSmallestVector();
268
269 //Translation
270
271 if (isSecondStatic) {
272 Translate(-depthPenetration);
273 }
274 else if (isFirstStatic) {
275 coll2->GetOwner()->GetComponentOfClass<C_PhysicsComponent>()->Translate(depthPenetration);
276 }
277 else {
278 depthPenetration = depthPenetration / 2.0f;
279 Translate(-depthPenetration);
280
281 coll2->GetOwner()->GetComponentOfClass<C_PhysicsComponent>()->Translate(depthPenetration);
282 }
283
284 //Force reflection
285 auto physicsComp = coll2->GetOwner()->GetComponentOfClass<C_PhysicsComponent>();
286
287 if (depthPenetration != FVector3(0.0f)) {
288
289 if (!isFirstStatic) {
290 FVector3 force =
291 (
292 (depthPenetration) * 2.0f *
293 (GetVelocity() * (depthPenetration))
294 / pow((depthPenetration).Length(), 2)
295 ) * -1.0f;
296
297 if (force != FVector3(0.0f)) {
298 SetVelocity((GetVelocity() + (force / GetMass())) * rubberness);
299 }
300 }
301
302
303 if (!isSecondStatic) {
304 FVector3 force2 =
305 (
306 (depthPenetration) * 2.0f *
307 (physicsComp->GetVelocity() * (depthPenetration))
308 / pow((depthPenetration).Length(), 2)
309 ) * -1.0f;
310
311 if (force2 != FVector3(0.0f)) {
312 physicsComp->SetVelocity((physicsComp->GetVelocity() + (force2 / physicsComp->GetMass())) * physicsComp->GetRubberness());
313 }
314 }
315 }
316
317
318 coll1->RefreshBox();
319 coll2->RefreshBox();
320
321 physicsComp = nullptr;
