121void CoreEngine::HandleEvents()
122{
123 SDL_Event event;
124 while (SDL_PollEvent(&event))
125 {
126 auto eventType = event.type;
127 int32_t keycode;
128
129 if (event.window.event == SDL_WINDOWEVENT_RESIZED)
130 {
131 EngineRenderer->FramebufferResizeCallback();
132 LevelGraph::GetInstance()->FrameBufferResizeCallback();
133 }
134
135 //If the event is key or button realted, set the keycode
136 switch (eventType)
137 {
138 case SDL_CONTROLLERBUTTONUP:
139 case SDL_CONTROLLERBUTTONDOWN:
140 keycode = event.cbutton.button;
141 break;
142 case SDL_KEYUP:
143 case SDL_KEYDOWN:
144 keycode = event.key.keysym.sym;
145 break;
146 case SDL_MOUSEBUTTONUP:
147 case SDL_MOUSEBUTTONDOWN:
148 keycode = event.button.button;
149 break;
150 default:
151 keycode = SDLK_UNKNOWN;
152 }
153 //Then find the function in the map using the event type and keycode as directions, if it exists, call it.
154 if (EngineInputFunctions[std::make_pair(eventType, keycode)])
155 {
156 EngineInputFunctions[std::make_pair(eventType, keycode)](&event);
157 }
158 }
159}
