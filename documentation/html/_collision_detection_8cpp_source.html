96bool CollisionDetection::RaySphereIntersection(Ray ray, Sphere sphere)
97{
98 FVector3 l = sphere.position - ray.GetOrigin();
99 float tca = l.GetDotProduct(ray.GetDirection());
100 if (tca < 0) { return false; }
101
102 float d2 = l.GetDotProduct(l) - tca * tca;
103 float radius2 = sphere.radius * sphere.radius;
104 if (d2 > radius2) { return false; }
105
106 float thc = sqrt(radius2 - d2);
107 float t0 = tca - thc;
108 float t1 = tca + thc;
109
110
111 if (t0 > t1) { std::swap(t0, t1); }
112
113 if (t0 < 0) {
114 t0 = t1; // if t0 is negative, let's use t1 instead
115 if (t0 < 0) return false; // both t0 and t1 are negative
116 }
117
118
119
120 ray.SetIntersectDistance(t0);
