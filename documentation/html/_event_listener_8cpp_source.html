17void EventListener::HandleEvents()
18{
19 SDL_Event event;
20 inputKey key;
21 while (SDL_PollEvent(&event))
22 {
23 switch (event.type)
24 {
25 case SDL_EventType::SDL_WINDOWEVENT:
26 if (event.window.event == SDL_WINDOWEVENT_SIZE_CHANGED)
27 {
28 CoreEngine::GetInstance()->GetRenderer()->FramebufferResizeCallback();
29 }
30 break;
31 case SDL_EventType::SDL_QUIT:
32 CoreEngine::GetInstance()->Quit(&event);
33 }
34
35 key.first = event.type;
36
37 //If the event is keyboard or button realted, set key.second
38 switch (key.first)
39 {
40 case SDL_CONTROLLERBUTTONUP:
41 case SDL_CONTROLLERBUTTONDOWN:
42 key.second = event.cbutton.button;
43 break;
44 case SDL_KEYUP:
45 case SDL_KEYDOWN:
46 key.second = event.key.keysym.sym;
47 break;
48 case SDL_MOUSEBUTTONUP:
49 case SDL_MOUSEBUTTONDOWN:
50 key.second = event.button.button;
51 case SDL_MOUSEMOTION:
52 case SDL_MOUSEWHEEL:
53 MouseEventHandler::UpdateCursorPosition();
54 break;
55 default:
56 key.second = SDLK_UNKNOWN;
57 break;
58 }
59 //Then find the function in the map using the event type and keycode as directions, if it exists, call it.
60 CallFunctions(key, &event);
61 }
