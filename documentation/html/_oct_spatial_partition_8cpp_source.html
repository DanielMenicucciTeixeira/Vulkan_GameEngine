260void OctSpatialPartition::UpdateColliderNode(C_BoundingBox* collider)
261{
262
263 //Trick to prevent need for double check.
264 int c1 = 0;
265 int c2 = 0;
266 bool doesExist = false;
267 std::set<int> copies;
268
269 auto newCells = GetCollidingNodes(collider->GetBoxBounds());
270 auto remainingCells = collider->GetCurrentNodes();
271
272 std::vector<OctNode*> currentCells;
273 currentCells.reserve(4);
274
275 //Check to see which one's exist in both, then remove them from each.
276 for(auto& cell : newCells){
277 for (auto& cell2 : remainingCells) {
278 if (cell == cell2) {
279 currentCells.push_back(cell2);
280 doesExist = true;
281 break;
282 }
283 c2++;
284 }
285 if (doesExist) {
286 remainingCells.erase(remainingCells.begin() + c2);
287 copies.insert(c1);
288 doesExist = false;
289 }
290 c2 = 0;
291 c1++;
292 }
293 int loop = 0;
294 for (auto cop : copies) {
295 newCells.erase(newCells.begin() + (cop - loop));
296 loop++;
297 }
298
299 //if new && remaining are both 0 return early
300 if (newCells.empty() && remainingCells.empty()) { return; }
301
302 //Add and remove remaining
303
304 for (auto& cell: newCells) {
305 cell->AddCollider(collider);
306 }
307 for (auto cell : remainingCells) {
308 cell->RemoveCollider(collider);
309 }
310
311 //merge both and set to collider.
312 for (auto& cell : currentCells) {
313 newCells.push_back(cell);
314 }
315
316 collider->SetCurrentNodes(newCells);
